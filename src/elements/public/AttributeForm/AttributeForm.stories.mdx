  {() => html`
    <foxy-attribute-form
      parent="https://demo.foxycart.com/s/admin/customers/0/attributes"
      class="container-narrow"
      @update=${((isUpdating = false) => evt => {
        if (!isUpdating && !evt.target.in({ idle: { template: { dirty: 'valid' } } })) {
          isUpdating = true;
          evt.target.data = null;
          evt.target.edit({ name: 'Foo', value: 'Bar' });
          evt.target.updateComplete.then(() => (isUpdating = false));
        }
      })()}
    >
    </foxy-attribute-form>
  `}
</Story>

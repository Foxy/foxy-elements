# foxy-payment-card-field

Form-associated hosted card field element that mounts a secure iframe for card entry or card security code entry and provides tokenization and validation APIs.

Required attributes: none. The secure embed origin is always taken from `VITE_EMBED_ORIGIN` at build time.

## Usage Example

```html
<form>
  <label for="card-field">Card details</label>
  <foxy-payment-card-field
    id="card-field"
    mode="card"
    lang="en-US"
    theme-input-height="44px"
    theme-input-padding="0 12px"
  ></foxy-payment-card-field>

  <button type="button" id="card-tokenize">Tokenize Card</button>
</form>

<script type="module">
  import "https://cdn-js.foxy.io/elements@2/foxy-payment-card-field.js";

  const cardField = document.querySelector("#card-field");
  const cardTokenizeButton = document.querySelector("#card-tokenize");

  cardField.addEventListener("tokenizationsuccess", (event) => {
    console.log("Card tokenized", event.detail.token, event.detail.requestId);
  });

  cardField.addEventListener("tokenizationerror", (event) => {
    console.error(
      "Card tokenization failed",
      event.detail.code,
      event.detail.message,
    );
  });

  cardTokenizeButton.addEventListener("click", async () => {
    const result = await cardField.tokenize();
    console.log("Card tokenize() result", result);
  });
</script>
```

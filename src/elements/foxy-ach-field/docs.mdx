# foxy-ach-field

Form-associated hosted ACH input element that renders one secure ACH field (routing number, account number, account type, or account holder name) and participates in grouped tokenization sessions.

Required attributes: none.

## Usage Example

```html
<form>
  <label for="ach-routing">Routing number</label>
  <foxy-ach-field
    id="ach-routing"
    type="routing-number"
    group="checkout-ach"
    placeholder="123456789"
    theme-input-height="44px"
  ></foxy-ach-field>

  <label for="ach-account">Account number</label>
  <foxy-ach-field
    id="ach-account"
    type="account-number"
    group="checkout-ach"
    placeholder="000123456789"
  ></foxy-ach-field>

  <button type="button" id="ach-tokenize">Tokenize ACH</button>
</form>

<script type="module">
  import "https://cdn-js.foxy.io/elements@2/foxy-ach-field.js";

  const achAccountField = document.querySelector("#ach-account");
  const achTokenizeButton = document.querySelector("#ach-tokenize");

  achAccountField.addEventListener("tokenizationsuccess", (event) => {
    console.log("ACH tokenized", event.detail.token, event.detail.requestId);
  });

  achAccountField.addEventListener("tokenizationerror", (event) => {
    console.error("ACH tokenization failed", event.detail.code);
  });

  achTokenizeButton.addEventListener("click", async () => {
    const result = await achAccountField.tokenize();
    console.log("ACH tokenize() result", result);
  });
</script>
```
